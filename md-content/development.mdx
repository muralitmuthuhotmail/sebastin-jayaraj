## Development Setup

### Prerequisites

Ensure you have the required tools installed:

```bash
# Check Node.js version (18+ required)
node --version

# Check pnpm version (10.7.0+ recommended)
pnpm --version

# Install if missing
npm install -g pnpm
```

### Getting Started

```bash
# Clone the repository
git clone https://github.com/muralitmuthuhotmail/findoora-docs.git
cd findoora-docs

# Install dependencies
pnpm install

# Start development server
pnpm dev

# Open http://localhost:3000
```

## Advanced Development

### Content Processing

#### Custom Content Transformers

```typescript
// lib/utils/content-processor.ts
import { remark } from "remark";
import remarkGfm from "remark-gfm";
import remarkMath from "remark-math";
import { rehype } from "rehype";
import rehypeSlug from "rehype-slug";
import rehypeAutolinkHeadings from "rehype-autolink-headings";

export interface ContentProcessor {
  processMarkdown(content: string): Promise<string>;
  extractMetadata(content: string): ContentMetadata;
  generateTableOfContents(content: string): TableOfContents[];
}

class MDXContentProcessor implements ContentProcessor {
  private remarkProcessor = remark().use(remarkGfm).use(remarkMath);

  private rehypeProcessor = rehype()
    .use(rehypeSlug)
    .use(rehypeAutolinkHeadings, {
      behavior: "wrap",
      properties: {
        className: ["anchor-link"],
        "aria-label": "Link to heading",
      },
    });

  async processMarkdown(content: string): Promise<string> {
    const remarkResult = await this.remarkProcessor.process(content);
    const rehypeResult = await this.rehypeProcessor.process(remarkResult);
    return String(rehypeResult);
  }

  extractMetadata(content: string): ContentMetadata {
    const frontmatterRegex = /^---\n([\s\S]*?)\n---/;
    const match = content.match(frontmatterRegex);

    if (!match) return {};

    const frontmatter = match[1];
    const metadata: ContentMetadata = {};

    frontmatter.split("\n").forEach((line) => {
      const [key, ...valueParts] = line.split(":");
      if (key && valueParts.length > 0) {
        const value = valueParts.join(":").trim();
        metadata[key.trim()] = this.parseValue(value);
      }
    });

    return metadata;
  }

  generateTableOfContents(content: string): TableOfContents[] {
    const headingRegex = /^(#{1,6})\s+(.+)$/gm;
    const toc: TableOfContents[] = [];
    let match;

    while ((match = headingRegex.exec(content)) !== null) {
      const level = match[1].length;
      const title = match[2].trim();
      const slug = title
        .toLowerCase()
        .replace(/\s+/g, "-")
        .replace(/[^\w-]/g, "");

      toc.push({
        level,
        title,
        slug,
        children: [],
      });
    }

    return this.buildTocHierarchy(toc);
  }

  private parseValue(value: string): any {
    // Parse arrays
    if (value.startsWith("[") && value.endsWith("]")) {
      return value
        .slice(1, -1)
        .split(",")
        .map((item) => item.trim().replace(/['"]/g, ""));
    }

    // Parse booleans
    if (value === "true") return true;
    if (value === "false") return false;

    // Parse strings
    return value.replace(/['"]/g, "");
  }

  private buildTocHierarchy(flatToc: TableOfContents[]): TableOfContents[] {
    const hierarchy: TableOfContents[] = [];
    const stack: TableOfContents[] = [];

    flatToc.forEach((item) => {
      // Find parent level
      while (stack.length > 0 && stack[stack.length - 1].level >= item.level) {
        stack.pop();
      }

      if (stack.length === 0) {
        hierarchy.push(item);
      } else {
        stack[stack.length - 1].children.push(item);
      }

      stack.push(item);
    });

    return hierarchy;
  }
}

export const contentProcessor = new MDXContentProcessor();
```

### Search Implementation

#### Advanced Search Features

```typescript
// lib/utils/search.ts
import Fuse from "fuse.js";

export interface SearchResult {
  item: ContentPost;
  score: number;
  matches: Fuse.FuseResultMatch[];
  highlights: string[];
}

export class AdvancedSearch {
  private fuse: Fuse<ContentPost>;
  private searchConfig: Fuse.IFuseOptions<ContentPost> = {
    includeScore: true,
    includeMatches: true,
    threshold: 0.3,
    minMatchCharLength: 2,
    keys: [
      { name: "title", weight: 5 },
      { name: "summary", weight: 3 },
      { name: "content", weight: 1 },
      { name: "tags", weight: 2 },
      { name: "category", weight: 2 },
    ],
  };

  constructor(posts: ContentPost[]) {
    this.fuse = new Fuse(posts, this.searchConfig);
  }

  search(query: string, filters?: SearchFilters): SearchResult[] {
    if (!query.trim()) return [];

    let results = this.fuse.search(query);

    // Apply filters
    if (filters) {
      results = this.applyFilters(results, filters);
    }

    // Transform results
    return results.map((result) => ({
      item: result.item,
      score: result.score || 0,
      matches: result.matches || [],
      highlights: this.generateHighlights(result),
    }));
  }

  private applyFilters(
    results: Fuse.FuseResult<ContentPost>[],
    filters: SearchFilters,
  ): Fuse.FuseResult<ContentPost>[] {
    return results.filter((result) => {
      if (
        filters.categories &&
        !filters.categories.includes(result.item.category)
      ) {
        return false;
      }

      if (
        filters.tags &&
        !filters.tags.some((tag) => result.item.tags?.includes(tag))
      ) {
        return false;
      }

      if (filters.dateRange) {
        const publishedDate = new Date(result.item.publishedAt);
        if (
          publishedDate < filters.dateRange.start ||
          publishedDate > filters.dateRange.end
        ) {
          return false;
        }
      }

      return true;
    });
  }

  private generateHighlights(result: Fuse.FuseResult<ContentPost>): string[] {
    const highlights: string[] = [];

    result.matches?.forEach((match) => {
      if (match.indices && match.value) {
        match.indices.forEach(([start, end]) => {
          const before = match.value!.substring(Math.max(0, start - 20), start);
          const highlighted = match.value!.substring(start, end + 1);
          const after = match.value!.substring(
            end + 1,
            Math.min(match.value!.length, end + 21),
          );

          highlights.push(`...${before}<mark>${highlighted}</mark>${after}...`);
        });
      }
    });

    return highlights;
  }
}
```

```tsx
// components/ui/optimized-image.tsx
import Image from "next/image";
import { useState } from "react";

interface OptimizedImageProps {

  height = 600,
  className,
  priority = false,

  const [isLoading, setIsLoading] = useState(true);
```
